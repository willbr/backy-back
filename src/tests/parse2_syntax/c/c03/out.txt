[
    [include-lib "stdio.h" ie/newline]
    [def main ie/newline
        [puts "hi" ie/newline]
        [c :=
            [ie/neoteric int
                [ie/infix 0]]
            ie/newline]
        [while
            [ie/infix c =
                [ie/neoteric getchar
                    [ie/infix]]]
            != EOF ie/newline
            [putchar c]]]]
