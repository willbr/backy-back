[
    [include-lib "stdio.h" ie/newline]
    [def main ie/newline
        [nc :=
            [ie/neoteric long
                [ie/infix 0]]
            ie/newline]
        [while
            [ie/neoteric getchar
                [ie/infix]]
            != EOF ie/newline
            [inc nc ie/newline]]
        [printf "%ld\n" nc]]]
